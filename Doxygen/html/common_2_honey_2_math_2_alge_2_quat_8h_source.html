55  // Creates an order value between 0 and 23 from 4-tuple choices
56  /*
57  * There are 24 possible conventions, designated by:
58  * - eulAxI, axis used initially
59  * - eulPar, parity of axis permutation
60  * - eulRep, repetition of initial axis as last
61  * - eulFrm, frame from which axes are taken
62  *
63  * Axes I,J,K will be a permutation of X,Y,Z.
64  * Axis H will be either I or K, depending on eulRep.
65  * Frame S takes axes from initial static frame.
66  * If ord = (AxI=X, Par=Even, Rep=No, Frm=S), then {a,b,c,ord} means Rz(c)Ry(b)Rx(a),
67  * where Rz(c)v rotates v around Z by c radians.
68  *
69  * Ken Shoemake, 1993
70  */

325  static Real slerpCorrection(Real t, Real cosAlpha)
326  {
327  static const Real tcor = 0.58549219;
328  static const Real tcoratten = 0.82279687;
329 
330  Real factor = 1 - tcoratten*cosAlpha;
