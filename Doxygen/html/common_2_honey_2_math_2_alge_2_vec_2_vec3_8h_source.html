106  tuple<Vec3,Vec3> orthonormalBasis() const
107  {
108  Vec u,v;
109  const Vec3& w = *this;
110 
111  if (Alge::abs(w[0]) >= Alge::abs(w[1]))
112  {
113  // w.x or w.z is the largest magnitude component, swap them
114  Real invLength = Alge::sqrtInv(w[0]*w[0] + w[2]*w[2]);
115  u[0] = -w[2]*invLength;
116  u[1] = 0;
117  u[2] = +w[0]*invLength;
118  v[0] = w[1]*u[2];
119  v[1] = w[2]*u[0] - w[0]*u[2];
120  v[2] = -w[1]*u[0];
121  }
122  else
123  {
124  // w.y or w.z is the largest magnitude component, swap them
125  Real invLength = Alge::sqrtInv(w[1]*w[1] + w[2]*w[2]);
126  u[0] = 0;
127  u[1] = +w[2]*invLength;
128  u[2] = -w[1]*invLength;
129  v[0] = w[1]*u[2] - w[2]*u[1];
130  v[1] = -w[0]*u[2];
131  v[2] = w[0]*u[1];
132  }
133  return make_tuple(u,v);
134  }
