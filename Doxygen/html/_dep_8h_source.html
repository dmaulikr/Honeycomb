508  void condense()
509  {
510  CondenseData data;
511  VertexLinkList linkList;
512 
513  for (auto& vertex : stdutil::values(_vertexMap))
514  {
515  assert(data.stackS.empty() && data.stackP.empty(), "Condense algorithm failure");
516  linkList.clear();
517  linkList.push_back(mtmap( k_vertex() = vertex,
518  k_linkMap() = &vertex->linkMap(DepType::out),
519  k_linkIt() = vertex->linkMap(DepType::out).begin() ));
520  condense(data, linkList);
521  }
522 
523  for (auto& e : data.mergeMap)
524  {
525  auto& mergeVertex = *e.first;
526  auto& oldVertexList = e.second;
527 
528  //Build merged vertex links
529  for (auto i : honey::range(DepType::valMax))
530  {
531  auto type = DepType(i);
532  auto depTypeOpp = DepNode::depTypeOpp(type);
533 
534  //Loop through old vertices
535  for (auto& e : oldVertexList)
536  {
537  auto& oldVertex = *e;
538  for (auto& e : oldVertex.linkMap(type))
539  {
540  auto& vertex = *e.first;
541 
542  //Ignore link to self
543  if (oldVertexList.count(&vertex)) continue;
544 
545  //If the vertex at the other end is also being merged then link to that merged vertex instead
546  auto pVertex = &vertex;
547  auto itMerged = data.mergeMapR.find(pVertex);
548  if (itMerged != data.mergeMapR.end()) pVertex = itMerged->second;
549 
550  //Add link to merged vertex
551  mergeVertex.add(type, pVertex, e.second);
552 
553  //Merge links at other end
554  //If the vertex at the other end is also being merged then its links are clean
555  if (pVertex != &vertex) continue;
556 
557  //Get old link
558  auto& linkMap = vertex.linkMap(depTypeOpp);
559  auto itOld = linkMap.find(&oldVertex);
560  assert(itOld != linkMap.end(), "Old link not found, problem with algorithm");
561  //Move old link into merged link
562  vertex.add(depTypeOpp, &mergeVertex, itOld->second);
563  linkMap.erase(itOld);
564  }
565  }
566  }
567 
568  //Remap to merged vertex
569  for (auto& e : mergeVertex.keyList) { _vertexMap[e] = &mergeVertex; }
570 
571  //Delete old vertices

573  {
574  auto& oldVertex = *e;
575  //Clear key list so delete doesn't unmap merged vertex
576  oldVertex.keyList.clear();
577  deleteVertex(oldVertex);

630  void condense(CondenseData& data, const VertexLinkList& linkList)
631  {
632  auto vertex = linkList.back()[k_vertex()];
633  if (!data.preOrdMap.insert(make_pair(vertex, data.preOrd)).second) return;
634  ++data.preOrd;
635  data.stackS.push_back(vertex);
636  data.stackP.push_back(vertex);
637 
638  VertexLinkList recurseList;
639 
640  for (auto& e : linkList)
641  {
642  auto links = honey::range(e[k_linkIt()], e[k_linkMap()]->end());
643  for (auto& linkVertex : stdutil::keys(links))
644  {
645  auto itPreOrd = data.preOrdMap.find(linkVertex);
646  if (itPreOrd == data.preOrdMap.end())
647  {
648  //Move along out links
649  recurseList.clear();
650  recurseList.push_back(mtmap(k_vertex() = linkVertex,
651  k_linkMap() = &linkVertex->linkMap(DepType::out),
652  k_linkIt() = linkVertex->linkMap(DepType::out).begin() ));
653  condense(data, recurseList);
654  }
655  else if (!data.assignedList.count(linkVertex))
656  {
657  auto linkPreOrd = itPreOrd->second;
658  while (data.preOrdMap.find(data.stackP.back())->second > linkPreOrd)
659  {
660  data.stackP.pop_back();
661  assert(!data.stackP.empty(), "Condense algorithm failure");
662  }
663  }
664  }
665  }
666 
667  if (data.stackP.back() == vertex)
668  {
669  //Only merge if there is more than one node in merge list
670  if (data.stackS.back() != vertex)
671  {
672  //Merge strongly connected vertices
673  auto& mergeVertex = createVertex();
674 
675  Vertex* assignVertex;
676  do
677  {
678  assignVertex = data.stackS.back();
679  data.stackS.pop_back();
680 
681  data.assignedList.insert(assignVertex);
682  data.mergeMap[&mergeVertex].insert(assignVertex);
683  data.mergeMapR[assignVertex] = &mergeVertex;
684 
685  //Merge nodes into one node
686 
687  //Merge references
688  for (auto& e : assignVertex->nodeList)
689  {
690  //Ensure that merged references are unique. If this assert fails then there is a problem with the class' algorithms.
691  assert(!mergeVertex.nodeList.count(e), sout() << "Duplicate reference during condense merge. Node: " << e->getKey());

693  }
694 
695  //Merge keys
696  for (auto& e : assignVertex->keyList)

698  //Ensure that merged keys are unique. If this assert fails then there is a problem with the class' algorithms.
699  assert(!mergeVertex.keyList.count(e), sout() << "Duplicate key during condense merge. Node: " << e);
700  mergeVertex.keyList.insert(e);
701  }
702 
703  } while (assignVertex != vertex);
704  }
705  else
706  {
707  //There was only one node in merge list, ignore it
708  data.assignedList.insert(data.stackS.back());
709  data.stackS.pop_back();
710  }

712  data.stackP.pop_back();

717  void decondense(Vertex& mergeVertex)
718  {
719  //Unmap merged vertex
720  for (auto& e : mergeVertex.keyList) { _vertexMap.erase(e); }
721  //Re-add component nodes
722  for (auto& e : mergeVertex.nodeList) { add(*e); }
723 
724  //Loop through links and update links on other side
725  for (auto i : honey::range(DepType::valMax))
726  {
727  auto type = DepType(i);
728  auto depTypeOpp = DepNode::depTypeOpp(type);
729 
730  for (auto& e : stdutil::keys(mergeVertex.linkMap(type)))
731  {
732  auto& vertex = *e;
733  //Remove link to merged vertex
734  vertex.linkMap(depTypeOpp).erase(&mergeVertex);
735  //Loop through nodes and re-add links that have the same type and reference a component vertex
736  for (auto& e : vertex.nodeList)
