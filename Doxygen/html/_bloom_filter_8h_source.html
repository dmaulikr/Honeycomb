30  inline tuple<Real,int> calcParams(int elemCount, Real errorProb)
31  {
32  // Optimal hash count (k) formula that minimizes error (p):
33  // m = -n*ln(p) / ln(2)^2
34  Real bitCount = -elemCount * Alge::log(errorProb) / Alge::sqr(Alge::log(2));
35  // k = ln(2) * m / n
36  int hashCount = Alge::ceil(Alge::log(2) * bitCount / elemCount);
37  return make_tuple(bitCount, hashCount);
38  }
