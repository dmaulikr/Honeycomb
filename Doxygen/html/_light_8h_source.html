84  virtual Real calcIntensity(Real distance) const
85  {
86  Real intensity = 0;
87  switch (_decayType)
88  {
89  case DecayType::none:
90  intensity = 1;
91  break;
92  case DecayType::inv:
93  intensity = _radius/distance;
94  break;
95  case DecayType::invSqr:
96  intensity = (_radius*_radius)/(distance*distance);
97  break;
98  }
99  return Alge::min(intensity, 1);
100  }
101 
103  virtual Real calcDistance(Real intensity) const
104  {
105  intensity = Alge::clamp(intensity, 0, 1);
106  switch (_decayType)
107  {
108  case DecayType::none:
109  return intensity == 1 ? Real(0) : Real_::inf;
110  case DecayType::inv:
111  return _radius/intensity;
112  case DecayType::invSqr:
113  return Alge::sqrt((_radius / intensity) * _radius);
114  }
115  return 0;
116  }
