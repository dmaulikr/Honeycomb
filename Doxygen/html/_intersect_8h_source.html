242 inline bool Intersect_<Real>::test(const Box& box1, const Box& box2)
243 {
244  return !(box1.max.x < box2.min.x || box1.min.x > box2.max.x
245  || box1.max.y < box2.min.y || box1.min.y > box2.max.y
246  || box1.max.z < box2.min.z || box1.min.z > box2.max.z);

249 template<class Real>
250 inline bool Intersect_<Real>::test(const Box& box, const Sphere& sphere)
251 {
252  Real d = 0;
253 
254  if (sphere.center.x < box.min.x)
255  d += Alge::sqr(sphere.center.x - box.min.x);
256  else if (sphere.center.x > box.max.x)
257  d += Alge::sqr(sphere.center.x - box.max.x);
258 
259  if (sphere.center.y < box.min.y)
260  d += Alge::sqr(sphere.center.y - box.min.y);
261  else if (sphere.center.y > box.max.y)
262  d += Alge::sqr(sphere.center.y - box.max.y);
263 
264  if (sphere.center.z < box.min.z)
265  d += Alge::sqr(sphere.center.z - box.min.z);
266  else if (sphere.center.z > box.max.z)
267  d += Alge::sqr(sphere.center.z - box.max.z);
268 
269  return d <= Alge::sqr(sphere.radius);
270 }
