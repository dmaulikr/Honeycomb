423  Matrix<s_cols, s_rows, Real> inverse(option<Real&> det = optnull) const
424  {
425  auto res = Matrix<s_cols, s_rows, Real>().resize(cols(), rows());
426 
427  if (rows() == cols() && rows() <= 3)
428  {
429  //Square
430  switch(rows())
431  {
432  case 1:
433  {
434  Real d = determinant();
435  if (det) det = d;
436  if (Alge::isNearZero(d))
437  {
438  if (det) det = 0;
439  return res.fromZero();
440  }
441  res(0,0) = 1 / d;
442  break;
443  }
444 
445  case 2:
446  {
447  Real d = determinant();
448  if (det) det = d;
449  if (Alge::isNearZero(d))
450  {
451  if (det) det = 0;
452  return res.fromZero();
453  }
454  res(0,0) = m(1,1);
455  res(0,1) = -m(0,1);
456  res(1,0) = -m(1,0);
457  res(1,1) = m(0,0);
458  res /= d;
459  break;
460  }
461 
462  case 3:
463  {
464  Real d = determinant3(res);
465  if (det) det = d;
466  if (Alge::isNearZero(d))
467  {
468  if (det) det = 0;
469  return res.fromZero();
470  }
471  res /= d;
472  break;
473  }
474  }
475  }
476  else
477  {
478  //Non-square or > 3x3
479  Svd svd(subc());
480  svd.inverse(res);
481  //Determinant is the product of all non-zero singular values (or 1 if all are zero)
482  if (det) det = reduce(svd.w(), Real(1), [](Real a, Real e) { return !Alge::isNearZero(e) ? a*e : a; });
483  }
484 
485  return res;
486  }

489  Real determinant() const
490  {
491  Real res = 0;
492 
493  if (rows() == cols())
494  {
495  //Square
496  switch(rows())
497  {
498  case 1: res = m(0,0); break;
499  case 2: res = m(0,0)*m(1,1) - m(1,0)*m(0,1); break;
500  case 3: { MatrixS tmp(*this); res = determinant3(tmp); } break;
501  default:
502  {
503  for (int j = 0; j < cols(); j++)
504  {
505  auto minor_ = minor(0, j);
506  res += (1 - j%2 - j%2) * m(0,j) * minor_.determinant();
507  }
508  break;
509  }
510  }
511  }
512  else
513  {
514  //Non-square
515  //Determinant is the product of all non-zero singular values (or 1 if all are zero)
516  Svd svd;
517  res = reduce(svd.calcValues(subc()).w(), Real(1), [](Real a, Real e) { return !Alge::isNearZero(e) ? a*e : a; });
518  }
519  return res;
520  }
