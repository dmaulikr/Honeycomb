394  void scan()
395  {
396  ThreadData& td = threadData();
397 
398  //Set trace to make sure ref == 0 is consistent across tlref check below
399  for (auto* delNode = td.delHead; delNode; delNode = delNode->next)
400  {
401  Node& node = *delNode->node;
402  if (node.ref == 0)
403  {
404  node.trace = true;
405  if (node.ref != 0)
406  node.trace = false;
407  }
408  }
409 
410  //Flag all del nodes that have a tlref so they are not reclaimed
411  for (int ti = 0; ti < _threadDataCount; ++ti)
412  {
413  ThreadData* tdata = _threadDataList[ti];
414  for (int i = 0; i < size(tdata->tlrefs); ++i)
415  {
416  Node* node = tdata->tlrefs[i];
417  if (node && node->id < size(td.delTlrefs))
418  td.delTlrefs[node->id] = true;
419  }
420  }
421 
422  //Reclaim nodes and build new list of del nodes that could not be reclaimed
423  typename ThreadData::DelNode* newDelHead = nullptr;
424  int newDelCount = 0;
425 
426  while (td.delHead)
427  {
428  auto* delNode = td.delHead;
429  td.delHead = delNode->next;
430  Node& node = *delNode->node;
431  if (node.ref == 0 && node.trace && !td.delTlrefs[node.id])
432  {
433  delNode->node = nullptr;
434  if (delNode->claim == 0)
435  {
436  _config.terminateNode(node, false);
437  //Free the node
438  td.delNodeFreeList.push_back(delNode);
439  td.nodeFreeList.push_back(&node);
440  recycleFree(td);
441  continue;
442  }
443  _config.terminateNode(node, true);
444  delNode->done = true;
445  delNode->node = &node;
446  }
447  td.delTlrefs[node.id] = false;
448  delNode->next = newDelHead;

458  void recycleFree(ThreadData& td)
459  {
460  // Only recycle if private free list is too large
461  if (size(td.nodeFreeList) <= td.nodeCount*2) return;
462  // Return a chunk of nodes (up to _threshClean amount). Returning in chunks reduces number of recycle scans.
463  int newSize = td.nodeCount*2 - _threshClean;
464  if (newSize < td.nodeCount) newSize = td.nodeCount;
465  while (size(td.nodeFreeList) > newSize)
466  {
467  Node& node = *td.nodeFreeList.back();
468  td.nodeFreeList.pop_back();
469  auto& rec = _threadDataList[node.threadId]->recycleBins[td.id];
470  node.recycleNext = nullptr;
471  if (rec.tail) rec.tail->recycleNext = &node; //Previous tail can be consumed immediately atfer recycle next is set
472  rec.tail = &node;
473  if (!rec.head) rec.head = &node; //Head is null only on first recycle
474  }
475  }

478  void recycleScan(ThreadData& td)
479  {
480  // Take only up to _threshClean recycled nodes so loop doesn't take too long
481  int newSize = td.nodeCount < _threshClean ? td.nodeCount : _threshClean;
482  for (int ti = 0; ti < _threadDataCount && size(td.nodeFreeList) < newSize; ++ti)
483  {
484  auto& rec = td.recycleBins[ti];
485  Node* node = rec.head;
486  if (!node) continue;
487  Node* next = static_cast<Node*>(node->recycleNext.load());
488  if (!next) continue; //If next is null then this is the tail, not allowed to consume tail
489  do
490  {
491  td.nodeFreeList.push_back(node);
492  node = next;
493  next = static_cast<Node*>(node->recycleNext.load());
494  } while (next && size(td.nodeFreeList) < newSize);
495  rec.head = node; //Update head to next unconsumed node
496  }
497  }
